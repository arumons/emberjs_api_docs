 15 /**
 16   @class
 17 
 18   An unordered collection of objects.
 19 
 20   A Set works a bit like an array except that its items are not ordered.
 21   You can create a set to efficiently test for membership for an object. You
 22   can also iterate through a set just like an array, even accessing objects
 23   by index, however there is no guarantee as to their order.
 24 
 25   Starting with Ember 2.0 all Sets are now observable since there is no
 26   added cost to providing this support.  Sets also do away with the more
 27   specialized Set Observer API in favor of the more generic Enumerable
 28   Observer API - which works on any enumerable object including both Sets and
 29   Arrays.
 30 
 31   ## Creating a Set
 32 
 33   You can create a set like you would most objects using
 34   `new Ember.Set()`.  Most new sets you create will be empty, but you can
 35   also initialize the set with some content by passing an array or other
 36   enumerable of objects to the constructor.
 37 
 38   Finally, you can pass in an existing set and the set will be copied. You
 39   can also create a copy of a set by calling `Ember.Set#copy()`.
 40 
 41       #js
 42       // creates a new empty set
 43       var foundNames = new Ember.Set();
 44 
 45       // creates a set with four names in it.
 46       var names = new Ember.Set(["Charles", "Tom", "Juan", "Alex"]); // :P
 47 
 48       // creates a copy of the names set.
 49       var namesCopy = new Ember.Set(names);
 50 
 51       // same as above.
 52       var anotherNamesCopy = names.copy();
 53 
 54   ## Adding/Removing Objects
 55 
 56   You generally add or remove objects from a set using `add()` or
 57   `remove()`. You can add any type of object including primitives such as
 58   numbers, strings, and booleans.
 59 
 60   Unlike arrays, objects can only exist one time in a set. If you call `add()`
 61   on a set with the same object multiple times, the object will only be added
 62   once. Likewise, calling `remove()` with the same object multiple times will
 63   remove the object the first time and have no effect on future calls until
 64   you add the object to the set again.
 65 
 66   NOTE: You cannot add/remove null or undefined to a set. Any attempt to do so
 67   will be ignored.
 68 
 69   In addition to add/remove you can also call `push()`/`pop()`. Push behaves
 70   just like `add()` but `pop()`, unlike `remove()` will pick an arbitrary
 71   object, remove it and return it. This is a good way to use a set as a job
 72   queue when you don't care which order the jobs are executed in.
 73 
 74   ## Testing for an Object
 75 
 76   To test for an object's presence in a set you simply call
 77   `Ember.Set#contains()`.
 78 
 79   ## Observing changes
 80 
 81   When using `Ember.Set`, you can observe the `"[]"` property to be
 82   alerted whenever the content changes.  You can also add an enumerable
 83   observer to the set to be notified of specific objects that are added and
 84   removed from the set.  See `Ember.Enumerable` for more information on
 85   enumerables.
 86 
 87   This is often unhelpful. If you are filtering sets of objects, for instance,
 88   it is very inefficient to re-filter all of the items each time the set
 89   changes. It would be better if you could just adjust the filtered set based
 90   on what was changed on the original set. The same issue applies to merging
 91   sets, as well.
 92 
 93   ## Other Methods
 94 
 95   `Ember.Set` primary implements other mixin APIs.  For a complete reference
 96   on the methods you will use with `Ember.Set`, please consult these mixins.
 97   The most useful ones will be `Ember.Enumerable` and
 98   `Ember.MutableEnumerable` which implement most of the common iterator
 99   methods you are used to on Array.
100 
101   Note that you can also use the `Ember.Copyable` and `Ember.Freezable`
102   APIs on `Ember.Set` as well.  Once a set is frozen it can no longer be
103   modified.  The benefit of this is that when you call frozenCopy() on it,
104   Ember will avoid making copies of the set.  This allows you to write
105   code that can know with certainty when the underlying set data will or
106   will not be modified.
107 
108   @extends Ember.Enumerable
109   @extends Ember.MutableEnumerable
110   @extends Ember.Copyable
111   @extends Ember.Freezable
112 
113   @since Ember 0.9
114 */
115 Ember.Set = Ember.CoreObject.extend(Ember.MutableEnumerable, Ember.Copyable, Ember.Freezable,

122   /**
123     This property will change as the number of objects in the set changes.
124 
125     @property Number
126     @default 0
127   */
128   length: 0,
129 
130   /**
131     Clears the set. This is useful if you want to reuse an existing set
132     without having to recreate it.
133 
134         var colors = new Ember.Set(["red", "green", "blue"]);
135         colors.length;  => 3
136         colors.clear();
137         colors.length;  => 0
138 
139     @returns {Ember.Set} An empty Set
140   */
141   clear: function() {
142     if (this.isFrozen) { throw new Error(Ember.FROZEN_ERROR); }
143     var len = get(this, 'length');
144     this.enumerableContentWillChange(len, 0);
145     set(this, 'length', 0);
146     this.enumerableContentDidChange(len, 0);
147     return this;
148   },
149 
150   /**
151     Returns true if the passed object is also an enumerable that contains the
152     same objects as the receiver.
153 
154         var colors = ["red", "green", "blue"],
155             same_colors = new Ember.Set(colors);
156         same_colors.isEqual(colors); => true
157         same_colors.isEqual(["purple", "brown"]); => false
158 
159     @param {Ember.Set} obj the other object.
160     @returns {Boolean}
161   */
162   isEqual: function(obj) {
163     // fail fast
164     if (!Ember.Enumerable.detect(obj)) return false;
165 
166     var loc = get(this, 'length');
167     if (get(obj, 'length') !== loc) return false;
168 
169     while(--loc >= 0) {
170       if (!obj.contains(this[loc])) return false;
171     }
172 
173     return true;
174   },
175 
176   /**
177     Adds an object to the set. Only non-null objects can be added to a set
178     and those can only be added once. If the object is already in the set or
179     the passed value is null this method will have no effect.
180 
181     This is an alias for `Ember.MutableEnumerable.addObject()`.
182 
183         var colors = new Ember.Set();
184         colors.add("blue");    => ["blue"]
185         colors.add("blue");    => ["blue"]
186         colors.add("red");     => ["blue", "red"]
187         colors.add(null);      => ["blue", "red"]
188         colors.add(undefined); => ["blue", "red"]
189 
190     @function
191     @param {Object} obj The object to add.
192     @returns {Ember.Set} The set itself.
193   */
194   add: Ember.alias('addObject'),
195 
196   /**
197     Removes the object from the set if it is found.  If you pass a null value
198     or an object that is already not in the set, this method will have no
199     effect. This is an alias for `Ember.MutableEnumerable.removeObject()`.
200 
201         var colors = new Ember.Set(["red", "green", "blue"]);
202         colors.remove("red");    => ["blue", "green"]
203         colors.remove("purple"); => ["blue", "green"]
204         colors.remove(null);     => ["blue", "green"]
205 
206     @function
207     @param {Object} obj The object to remove
208     @returns {Ember.Set} The set itself.
209   */
210   remove: Ember.alias('removeObject'),
211 
212   /**
213     Removes the last element from the set and returns it, or null if it's empty.
214     
215         var colors = new Ember.Set(["green", "blue"]);
216         colors.pop(); => "blue"
217         colors.pop(); => "green"
218         colors.pop(); => null
219 
220     @returns {Object} The removed object from the set or null.
221   */
222   pop: function() {
223     if (get(this, 'isFrozen')) throw new Error(Ember.FROZEN_ERROR);
224     var obj = this.length > 0 ? this[this.length-1] : null;
225     this.remove(obj);
226     return obj;
227   },
228 
229   /**
230     Inserts the given object on to the end of the set. It returns
231     the set itself.
232 
233     This is an alias for `Ember.MutableEnumerable.addObject()`.
234 
235         var colors = new Ember.Set();
236         colors.push("red");   => ["red"]
237         colors.push("green"); => ["red", "green"]
238         colors.push("blue");  => ["red", "green", "blue"]
239 
240     @function
241     @returns {Ember.Set} The set itself.
242   */
243   push: Ember.alias('addObject'),
244 
245   /**
246     Removes the last element from the set and returns it, or null if it's empty.
247 
248     This is an alias for `Ember.Set.pop()`.
249 
250         var colors = new Ember.Set(["green", "blue"]);
251         colors.shift(); => "blue"
252         colors.shift(); => "green"
253         colors.shift(); => null
254 
255     @function
256     @returns {Object} The removed object from the set or null.
257   */
258   shift: Ember.alias('pop'),
259 
260   /**
261     Inserts the given object on to the end of the set. It returns
262     the set itself.
263 
264     This is an alias of `Ember.Set.push()`
265 
266         var colors = new Ember.Set();
267         colors.unshift("red");   => ["red"]
268         colors.unshift("green"); => ["red", "green"]
269         colors.unshift("blue");  => ["red", "green", "blue"]
270 
271     @function
272     @returns {Ember.Set} The set itself.
273   */
274   unshift: Ember.alias('push'),
275 
276   /**
277     Adds each object in the passed enumerable to the set.
278 
279     This is an alias of `Ember.MutableEnumerable.addObjects()`
280 
281         var colors = new Ember.Set();
282         colors.addEach(["red", "green", "blue"]); => ["red", "green", "blue"]
283 
284     @function
285     @param {Ember.Enumerable} objects the objects to add.
286     @returns {Ember.Set} The set itself.
287   */
288   addEach: Ember.alias('addObjects'),
289 
290   /**
291     Removes each object in the passed enumerable to the set.
292 
293     This is an alias of `Ember.MutableEnumerable.removeObjects()`
294 
295         var colors = new Ember.Set(["red", "green", "blue"]);
296         colors.removeEach(["red", "blue"]); => ["green"]
297 
298     @function
299     @param {Ember.Enumerable} objects the objects to remove.
300     @returns {Ember.Set} The set itself.
301 302   */
303   removeEach: Ember.alias('removeObjects'),

309   /** @private */
310 311   init: function(items) {
312     this._super();
313     if (items) this.addObjects(items);
314   },
315 
316   /** @private (nodoc) - implement Ember.Enumerable */
317   nextObject: function(idx) {
318     return this[idx];
319   },
320 
321   /** @private - more optimized version */
322   firstObject: Ember.computed(function() {
323     return this.length > 0 ? this[0] : undefined;
324   }).property('[]').cacheable(),
325 
326   /** @private - more optimized version */
327   lastObject: Ember.computed(function() {
328     return this.length > 0 ? this[this.length-1] : undefined;
329   }).property('[]').cacheable(),
330 
331   /** @private (nodoc) - implements Ember.MutableEnumerable */
332   addObject: function(obj) {
333     if (get(this, 'isFrozen')) throw new Error(Ember.FROZEN_ERROR);
334     if (none(obj)) return this; // nothing to do
335 
336     var guid = guidFor(obj),
337         idx  = this[guid],
338         len  = get(this, 'length'),
339         added ;
340 
341     if (idx>=0 && idx<len && (this[idx] === obj)) return this; // added
342 
343     added = [obj];
344     this.enumerableContentWillChange(null, added);
345     len = get(this, 'length');
346     this[guid] = len;
347     this[len] = obj;
348     set(this, 'length', len+1);
349     this.enumerableContentDidChange(null, added);
350 
351     return this;
352   },
353 
354   /** @private (nodoc) - implements Ember.MutableEnumerable */
355   removeObject: function(obj) {
356     if (get(this, 'isFrozen')) throw new Error(Ember.FROZEN_ERROR);
357     if (none(obj)) return this; // nothing to do
358 
359     var guid = guidFor(obj),
360         idx  = this[guid],
361         len = get(this, 'length'),
362         last, removed;
363 
364 
365     if (idx>=0 && idx<len && (this[idx] === obj)) {
366       removed = [obj];
367 
368       this.enumerableContentWillChange(removed, null);
369 
370       // swap items - basically move the item to the end so it can be removed
371       if (idx < len-1) {
372         last = this[len-1];
373         this[idx] = last;
374         this[guidFor(last)] = idx;
375       }
376 
377       delete this[guid];
378       delete this[len-1];
379       set(this, 'length', len-1);
380 
381       this.enumerableContentDidChange(removed, null);
382     }
383 
384     return this;
385   },
386 
387   /** @private (nodoc) - optimized version */
388   contains: function(obj) {
389     return this[guidFor(obj)]>=0;
390   },
391 
392   /** @private (nodoc) */
393   copy: function() {
394     var C = this.constructor, ret = new C(), loc = get(this, 'length');
395     set(ret, 'length', loc);
396     while(--loc>=0) {
397       ret[loc] = this[loc];
398       ret[guidFor(this[loc])] = loc;
399     }
400     return ret;
401   },
402 
403   /** @private */
404   toString: function() {
405     var len = this.length, idx, array = [];
406     for(idx = 0; idx < len; idx++) {
407       array[idx] = this[idx];
408     }
409     return "Ember.Set<%@>".fmt(array.join(','));
410   },

435 // Support the older API
436 var o_create = Ember.Set.create;
437 Ember.Set.create = function(items) {
438   if (items && Ember.Enumerable.detect(items)) {
439     ember_deprecate('Passing an enumerable to Ember.Set.create() is deprecated and will be removed in a future version of Ember.  Use new Ember.Set(items) instead.');
440     return new Ember.Set(items);
441   } else {
442     return o_create.apply(this, arguments);
443   }
