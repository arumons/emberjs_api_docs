1 /**
  2   JavaScript (before ES6) does not have a Map implementation. Objects,
  3   which are often used as dictionaries, may only have Strings as keys.
  4 
  5   Because Ember has a way to get a unique identifier for every object
  6   via `Ember.guidFor`, we can implement a performant Map with arbitrary
  7   keys. Because it is commonly used in low-level bookkeeping, Map is
  8   implemented as a pure JavaScript object for performance.
  9 
 10   This implementation follows the current iteration of the ES6 proposal
 11   for maps (http://wiki.ecmascript.org/doku.php?id=harmony:simple_maps_and_sets),
 12   with two exceptions. First, because we need our implementation to be
 13   pleasant on older browsers, we do not use the `delete` name (using
 14   `remove` instead). Second, as we do not have the luxury of in-VM
 15   iteration, we implement a forEach method for iteration.
 16 
 17   Map is mocked out to look like an Ember object, so you can do
 18   `Ember.Map.create()` for symmetry with other Ember classes.
 19 */
 20 
 21 var guidFor = Ember.guidFor;
 22 var indexOf = Ember.ArrayUtils.indexOf;
 23 
 24 // This class is used internally by Ember.js and Ember Data.
 25 // Please do not use it at this time. We plan to clean it up
 26 // and add many tests soon.
 27 var OrderedSet = Ember.OrderedSet = function() {
 28   this.clear();
 29 };
 30 
 31 OrderedSet.create = function() {
 32   return new OrderedSet();
 33 };
 34 
 35 OrderedSet.prototype = {
 36   clear: function() {
 37     this.presenceSet = {};
 38     this.list = [];
 39   },
 40 
 41   add: function(obj) {
 42     var guid = guidFor(obj),
 43         presenceSet = this.presenceSet,
 44         list = this.list;
 45 
 46     if (guid in presenceSet) { return; }
 47 
 48     presenceSet[guid] = true;
 49     list.push(obj);
 50   },
 51 
 52   remove: function(obj) {
 53     var guid = guidFor(obj),
 54         presenceSet = this.presenceSet,
 55         list = this.list;
 56 
 57     delete presenceSet[guid];
 58 
 59     var index = indexOf(list, obj);
 60     if (index > -1) {
 61       list.splice(index, 1);
 62     }
 63   },
 64 
 65   isEmpty: function() {
 66     return this.list.length === 0;
 67   },
 68 
 69   forEach: function(fn, self) {
 70     // allow mutation during iteration
 71     var list = this.list.slice();
 72 
 73     for (var i = 0, j = list.length; i < j; i++) {
 74       fn.call(self, list[i]);
 75     }
 76   },
 77 
 78   toArray: function() {
 79     return this.list.slice();
 80   }
 81 };
 82 
 83 /**
 84   A Map stores values indexed by keys. Unlike JavaScript's
 85   default Objects, the keys of a Map can be any JavaScript
 86   object.
 87 
 88   Internally, a Map has two data structures:
 89 
 90     `keys`: an OrderedSet of all of the existing keys
 91     `values`: a JavaScript Object indexed by the
 92       Ember.guidFor(key)
 93 
 94   When a key/value pair is added for the first time, we
 95   add the key to the `keys` OrderedSet, and create or
 96   replace an entry in `values`. When an entry is deleted,
 97   we delete its entry in `keys` and `values`.
 98 */
 99 
100 var Map = Ember.Map = function() {
101   this.keys = Ember.OrderedSet.create();
102   this.values = {};
103 };
104 
105 Map.create = function() {
106   return new Map();
107 };
108 
109 Map.prototype = {
110   /**
111     Retrieve the value associated with a given key.
112 
113     @param {anything} key
114     @return {anything} the value associated with the key, or undefined
115   */
116   get: function(key) {
117     var values = this.values,
118         guid = guidFor(key);
119 
120     return values[guid];
121   },
122 
123   /**
124     Adds a value to the map. If a value for the given key has already been
125     provided, the new value will replace the old value.
126 
127     @param {anything} key
128     @param {anything} value
129   */
130   set: function(key, value) {
131     var keys = this.keys,
132         values = this.values,
133         guid = guidFor(key);
134 
135     keys.add(key);
136     values[guid] = value;
137   },
138 
139   /**
140     Removes a value from the map for an associated key.
141 
142     @param {anything} key
143     @returns {Boolean} true if an item was removed, false otherwise
144   */
145   remove: function(key) {
146     // don't use ES6 "delete" because it will be annoying
147     // to use in browsers that are not ES6 friendly;
148     var keys = this.keys,
149         values = this.values,
150         guid = guidFor(key),
151         value;
152 
153     if (values.hasOwnProperty(guid)) {
154       keys.remove(key);
155       value = values[guid];
156       delete values[guid];
157       return true;
158     } else {
159       return false;
160     }
161   },
162 
163   /**
164     Check whether a key is present.
165 
166     @param {anything} key
167     @returns {Boolean} true if the item was present, false otherwise
168   */
169   has: function(key) {
170     var values = this.values,
171         guid = guidFor(key);
172 
173     return values.hasOwnProperty(guid);
174   },
175 
176   /**
177     Iterate over all the keys and values. Calls the function once
178     for each key, passing in the key and value, in that order.
179 
180     The keys are guaranteed to be iterated over in insertion order.
181 
182     @param {Function} callback
183     @param {anything} self if passed, the `this` value inside the
184       callback. By default, `this` is the map.
185   */
186   forEach: function(callback, self) {
187     var keys = this.keys,
188         values = this.values;
189 
190     keys.forEach(function(key) {
191       var guid = guidFor(key);
192       callback.call(self, key, values[guid]);
193     });
194   }
195 };
