 77 // Coerces a non-array value into an array.
 78 /** @private */
 79 function MULTIPLE(val) {
 80   if (val instanceof Array) return val;
 81   if (val === undefined || val === null) return [];
 82   return [val];
 83 }
 84 
 85 // Treats a single-element array as the element. Otherwise
 86 // returns a placeholder.
 87 /** @private */
 88 function SINGLE(val, placeholder) {
 89   if (val instanceof Array) {
 90     if (val.length>1) return placeholder;
 91     else return val[0];
 92   }
 93   return val;
 94 }

117 // Applies a binding's transformations against a value.
118 /** @private */
119 function getTransformedValue(binding, val, obj, dir) {
120 
121   // First run a type transform, if it exists, that changes the fundamental
122   // type of the value. For example, some transforms convert an array to a
123   // single object.
124 
125   var typeTransform = binding._typeTransform;
126   if (typeTransform) { val = typeTransform(val, binding._placeholder); }
127 
128   // handle transforms
129   var transforms = binding._transforms,
130       len        = transforms ? transforms.length : 0,
131       idx;
132 
133   for(idx=0;idx<len;idx++) {
134     var transform = transforms[idx][dir];
135     if (transform) { val = transform.call(this, val, obj); }
136   }
137   return val;
138 }

142   return val===undefined || val===null || val==='' || (Ember.isArray(val) && get(val, 'length')===0) ;

145 /** @private */
146 function getPathWithGlobals(obj, path) {
147   return getPath(isGlobalPath(path) ? window : obj, path);

150 /** @private */
151 function getFromValue(obj, binding) {
152   var operation = binding._operation;
153 
154   if (operation) {
155     return operation(obj, binding._from, binding._operand);
156   } else {
157     return getPathWithGlobals(obj, binding._from);
158   }
159 }
160 
161 /** @private */
162 function getToValue(obj, binding) {
163   return getPath(obj, binding._to);
164 }

166 /** @private */
167 var AND_OPERATION = function(obj, left, right) {
168   return getPathWithGlobals(obj, left) && getPathWithGlobals(obj, right);
169 };
170 
171 /** @private */
172 var OR_OPERATION = function(obj, left, right) {
173   return getPathWithGlobals(obj, left) || getPathWithGlobals(obj, right);
174 };

429   /**
430     Attempts to connect this binding instance so that it can receive and relay
431     changes. This method will raise an exception if you have not set the
432     from/to properties yet.
433 
434     @param {Object} obj
435       The root object for this binding.
436 437 
    @param {Boolean} preferFromParam
438       private: Normally, `connect` cannot take an object if `from` already set
439       an object. Internally, we would like to be able to provide a default object
440       to be used if no object was provided via `from`, so this parameter turns
441       off the assertion.
442 
443     @returns {Ember.Binding} this
444   */
445   connect: function(obj) {
446     ember_assert('Must pass a valid object to Ember.Binding.connect()', !!obj);
447 
448     var oneWay = this._oneWay, operand = this._operand;
449 
450     // add an observer on the object to be notified when the binding should be updated
451     Ember.addObserver(obj, this._from, this, this.fromDidChange);
452 
453     // if there is an operand, add an observer onto it as well
454     if (operand) { Ember.addObserver(obj, operand, this, this.fromDidChange); }
455 
456     // if the binding is a two-way binding, also set up an observer on the target
457     // object.
458     if (!oneWay) { Ember.addObserver(obj, this._to, this, this.toDidChange); }
459 
460     if (Ember.meta(obj,false).proto !== obj) { this._scheduleSync(obj, 'fwd'); }
461 
462     this._readyToSync = true;
463     return this;
464   },
465 
466   /**
467     Disconnects the binding instance. Changes will no longer be relayed. You
468     will not usually need to call this method.
469 
470     @param {Object} obj
471       The root object you passed when connecting the binding.
472 
473     @returns {Ember.Binding} this
474   */
475   disconnect: function(obj) {
476     ember_assert('Must pass a valid object to Ember.Binding.disconnect()', !!obj);
477 
478     var oneWay = this._oneWay, operand = this._operand;
479 
480     // remove an observer on the object so we're no longer notified of
481     // changes that should update bindings.
482     Ember.removeObserver(obj, this._from, this, this.fromDidChange);
483 
484     // if there is an operand, remove the observer from it as well
485     if (operand) Ember.removeObserver(obj, operand, this, this.fromDidChange);
486 
487     // if the binding is two-way, remove the observer from the target as well
488     if (!oneWay) Ember.removeObserver(obj, this._to, this, this.toDidChange);
489 
490     this._readyToSync = false; // disable scheduled syncs...
491     return this;
492   },

508   /** @private */
509   _scheduleSync: function(obj, dir) {
510     var guid = guidFor(obj), existingDir = this[guid];
511 
512     // if we haven't scheduled the binding yet, schedule it
513     if (!existingDir) {
514       Ember.run.schedule('sync', this, this._sync, obj);
515       this[guid] = dir;
516     }
517 
518     // If both a 'back' and 'fwd' sync have been scheduled on the same object,
519     // default to a 'fwd' sync so that it remains deterministic.
520     if (existingDir === 'back' && dir === 'fwd') {
521       this[guid] = 'fwd';
522     }
523   },
524 
525   /** @private */
526   _sync: function(obj) {
527     var log = Ember.LOG_BINDINGS;
528 
529     // don't synchronize destroyed objects or disconnected bindings
530     if (obj.isDestroyed || !this._readyToSync) { return; }
531 
532     // get the direction of the binding for the object we are
533     // synchronizing from
534     var guid = guidFor(obj), direction = this[guid];
535 
536     var fromPath = this._from, toPath = this._to, lastSet;
537 
538     delete this[guid];
539 
540     if (direction === 'fwd') {
541       lastSet = this._cache.back;
542     } else if (direction === 'back') {
543       lastSet = this._cache.fwd;
544     }
545 
546     var fromValue, toValue;
547 
548     // There's a bit of duplicate logic here, but the order is important.
549     //
550     // We want to avoid ping-pong bindings. To do this, we store off the
551     // guid of the item we are setting. Later, we avoid synchronizing
552     // bindings in the other direction if the raw value we are copying
553     // is the same as the guid of the last thing we set.
554     //
555     // Use guids here to avoid unnecessarily holding hard references
556     // to objects.
557     if (direction === 'fwd') {
558       fromValue = getFromValue(obj, this);
559       if (this._cache.back === guidFor(fromValue)) { return; }
560       this._cache.fwd = guidFor(fromValue);
561 
562       toValue = getToValue(obj, this);
563     } else if (direction === 'back') {
564       toValue = getToValue(obj, this);
565       if (this._cache.fwd === guidFor(toValue)) { return; }
566       this._cache.back = guidFor(toValue);
567 
568       fromValue = getFromValue(obj, this);
569     }
570 
571     fromValue = getTransformedValue(this, fromValue, obj, 'to');
572     toValue = getTransformedValue(this, toValue, obj, 'from');
573 
574     if (toValue === fromValue) { return; }
575 
576     // if we're synchronizing from the remote object...
577     if (direction === 'fwd') {
578       if (log) { Ember.Logger.log(' ', this.toString(), toValue, '->', fromValue, obj); }
579       Ember.trySetPath(Ember.isGlobalPath(toPath) ? window : obj, toPath, fromValue);
580 
581     // if we're synchronizing *to* the remote object
582     } else if (direction === 'back') {// && !this._oneWay) {
583       if (log) { Ember.Logger.log(' ', this.toString(), toValue, '<-', fromValue, obj); }
584       Ember.trySetPath(Ember.isGlobalPath(fromPath) ? window : obj, fromPath, toValue);
585     }
586   }

719 /**
720   @class
721 
722   A binding simply connects the properties of two objects so that whenever the
723   value of one property changes, the other property will be changed also. You
724   do not usually work with Binding objects directly but instead describe
725   bindings in your class definition using something like:
726 
727         valueBinding: "MyApp.someController.title"
728 
729   This will create a binding from `MyApp.someController.title` to the `value`
730   property of your object instance automatically. Now the two values will be
731   kept in sync.
732 
733   ## Customizing Your Bindings
734 
735   In addition to synchronizing values, bindings can also perform some basic
736   transforms on values. These transforms can help to make sure the data fed
737   into one object always meets the expectations of that object regardless of
738   what the other object outputs.
739 
740   To customize a binding, you can use one of the many helper methods defined
741   on Ember.Binding like so:
742 
743         valueBinding: Ember.Binding.single("MyApp.someController.title")
744 
745   This will create a binding just like the example above, except that now the
746   binding will convert the value of `MyApp.someController.title` to a single
747   object (removing any arrays) before applying it to the `value` property of
748   your object.
749 
750   You can also chain helper methods to build custom bindings like so:
751 
752         valueBinding: Ember.Binding.single("MyApp.someController.title").notEmpty("(EMPTY)")
753 
754   This will force the value of MyApp.someController.title to be a single value
755   and then check to see if the value is "empty" (null, undefined, empty array,
756   or an empty string). If it is empty, the value will be set to the string
757   "(EMPTY)".
758 
759   ## One Way Bindings
760 
761   One especially useful binding customization you can use is the `oneWay()`
762   helper. This helper tells Ember that you are only interested in
763   receiving changes on the object you are binding from. For example, if you
764   are binding to a preference and you want to be notified if the preference
765   has changed, but your object will not be changing the preference itself, you
766   could do:
767 
768         bigTitlesBinding: Ember.Binding.oneWay("MyApp.preferencesController.bigTitles")
769 
770   This way if the value of MyApp.preferencesController.bigTitles changes the
771   "bigTitles" property of your object will change also. However, if you
772   change the value of your "bigTitles" property, it will not update the
773   preferencesController.
774 
775   One way bindings are almost twice as fast to setup and twice as fast to
776   execute because the binding only has to worry about changes to one side.
777 
778   You should consider using one way bindings anytime you have an object that
779   may be created frequently and you do not intend to change a property; only
780   to monitor it for changes. (such as in the example above).
781 
782   ## Adding Custom Transforms
783 
784   In addition to using the standard helpers provided by Ember, you can
785   also defined your own custom transform functions which will be used to
786   convert the value. To do this, just define your transform function and add
787   it to the binding with the transform() helper. The following example will
788   not allow Integers less than ten. Note that it checks the value of the
789   bindings and allows all other values to pass:
790 
791         valueBinding: Ember.Binding.transform(function(value, binding) {
792           return ((Ember.typeOf(value) === 'number') && (value < 10)) ? 10 : value;
793         }).from("MyApp.someController.value")
794 
795   If you would like to instead use this transform on a number of bindings,
796   you can also optionally add your own helper method to Ember.Binding. This
797   method should simply return the value of `this.transform()`. The example
798   below adds a new helper called `notLessThan()` which will limit the value to
799   be not less than the passed minimum:
800 
801       Ember.Binding.reopen({
802         notLessThan: function(minValue) {
803           return this.transform(function(value, binding) {
804             return ((Ember.typeOf(value) === 'number') && (value < minValue)) ? minValue : value;
805           });
806         }
807       });
808 
809   You could specify this in your core.js file, for example. Then anywhere in
810   your application you can use it to define bindings like so:
811 
812         valueBinding: Ember.Binding.from("MyApp.someController.value").notLessThan(10)
813 
814   Also, remember that helpers are chained so you can use your helper along
815   with any other helpers. The example below will create a one way binding that
816   does not allow empty values or values less than 10:
817 
818         valueBinding: Ember.Binding.oneWay("MyApp.someController.value").notEmpty().notLessThan(10)
819 
820   Finally, it's also possible to specify bi-directional transforms. To do this,
821   you can pass a hash to `transform` with `to` and `from`. In the following
822   example, we are expecting a lowercase string that we want to transform to
823   uppercase.
824 
825         valueBinding: Ember.Binding.transform({
826           to:   function(value, binding) { return value.toUpperCase(); },
827           from: function(value, binding) { return value.toLowerCase(); }
828 
829   ## How to Manually Adding Binding
830 
831   All of the examples above show you how to configure a custom binding, but
832   the result of these customizations will be a binding template, not a fully
833   active binding. The binding will actually become active only when you
834   instantiate the object the binding belongs to. It is useful however, to
835   understand what actually happens when the binding is activated.
836 
837   For a binding to function it must have at least a "from" property and a "to"
838   property. The from property path points to the object/key that you want to
839   bind from while the to path points to the object/key you want to bind to.
840 
841   When you define a custom binding, you are usually describing the property
842   you want to bind from (such as "MyApp.someController.value" in the examples
843   above). When your object is created, it will automatically assign the value
844   you want to bind "to" based on the name of your binding key. In the
845   examples above, during init, Ember objects will effectively call
846   something like this on your binding:
847 
848         binding = Ember.Binding.from(this.valueBinding).to("value");
849 
850   This creates a new binding instance based on the template you provide, and
851   sets the to path to the "value" property of the new object. Now that the
852   binding is fully configured with a "from" and a "to", it simply needs to be
853   connected to become active. This is done through the connect() method:
854 
855         binding.connect(this);
856 
857   Note that when you connect a binding you pass the object you want it to be
858   connected to.  This object will be used as the root for both the from and
859   to side of the binding when inspecting relative paths.  This allows the
860   binding to be automatically inherited by subclassed objects as well.
861 
862   Now that the binding is connected, it will observe both the from and to side
863   and relay changes.
864 
865   If you ever needed to do so (you almost never will, but it is useful to
866   understand this anyway), you could manually create an active binding by
867   using the Ember.bind() helper method. (This is the same method used by
868   to setup your bindings on objects):
869 
870         Ember.bind(MyApp.anotherObject, "value", "MyApp.someController.value");
871 
872   Both of these code fragments have the same effect as doing the most friendly
873   form of binding creation like so:
874 
875         MyApp.anotherObject = Ember.Object.create({
876           valueBinding: "MyApp.someController.value",
877 
878           // OTHER CODE FOR THIS OBJECT...
879 
880         });
881 
882   Ember's built in binding creation method makes it easy to automatically
883   create bindings for you. You should always use the highest-level APIs
884   available, even if you understand how to it works underneath.
885 
886   @since Ember 0.9
887 */
888 Ember.Binding = Binding;
889 
890 /**
891   Global helper method to create a new binding.  Just pass the root object
892   along with a to and from path to create and connect the binding.  The new
893   binding object will be returned which you can further configure with
894   transforms and other conditions.
895 
896   @param {Object} obj
897     The root object of the transform.
898 
899   @param {String} to
900     The path to the 'to' side of the binding.  Must be relative to obj.
901 
902   @param {String} from
903     The path to the 'from' side of the binding.  Must be relative to obj or
904     a global path.
905 
906   @returns {Ember.Binding} binding instance
907 */
908 Ember.bind = function(obj, to, from) {
909   return new Ember.Binding(to, from).connect(obj);
910 };
911 
912 Ember.oneWay = function(obj, to, from) {
913   return new Ember.Binding(to, from).oneWay().connect(obj);
914 };
