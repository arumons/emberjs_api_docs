394   /**
395     @private
396 
397     Iterates over the view's `classNameBindings` array, inserts the value
398     of the specified property into the `classNames` array, then creates an
399     observer to update the view's element if the bound property ever changes
400     in the future.
401   */
402 403   _applyClassNameBindings: function() {
404     var classBindings = get(this, 'classNameBindings'),
405         classNames = get(this, 'classNames'),
406         elem, newClass, dasherizedClass;
407 
408     if (!classBindings) { return; }
409 
410     // Loop through all of the configured bindings. These will be either
411     // property names ('isUrgent') or property paths relative to the view
412     // ('content.isUrgent')
413     a_forEach(classBindings, function(binding) {
414 
415       // Variable in which the old class value is saved. The observer function
416       // closes over this variable, so it knows which string to remove when
417       // the property changes.
418       var oldClass, property;
419 
420       // Set up an observer on the context. If the property changes, toggle the
421       // class name.
422       var observer = function() {
423         // Get the current value of the property
424         newClass = this._classStringForProperty(binding);
425         elem = this.$();
426 
427         // If we had previously added a class to the element, remove it.
428         if (oldClass) {
429           elem.removeClass(oldClass);
430           // Also remove from classNames so that if the view gets rerendered,
431           // the class doesn't get added back to the DOM.
432           classNames.removeObject(oldClass);
433         }
434 
435         // If necessary, add a new class. Make sure we keep track of it so
436         // it can be removed in the future.
437         if (newClass) {
438           elem.addClass(newClass);
439           oldClass = newClass;
440         } else {
441           oldClass = null;
442         }
443       };
444 
445       // Get the class name for the property at its current value
446       dasherizedClass = this._classStringForProperty(binding);
447 
448       if (dasherizedClass) {
449         // Ensure that it gets into the classNames array
450         // so it is displayed when we render.
451         classNames.push(dasherizedClass);
452 
453         // Save a reference to the class name so we can remove it
454         // if the observer fires. Remember that this variable has
455         // been closed over by the observer.
456         oldClass = dasherizedClass;
457       }
458 
459       // Extract just the property name from bindings like 'foo:bar'
460       property = binding.split(':')[0];
461       addObserver(this, property, observer);
462     }, this);
463   },

465   /**
466     Iterates through the view's attribute bindings, sets up observers for each,
467     then applies the current value of the attributes to the passed render buffer.
468 
469     @param {Ember.RenderBuffer} buffer
470   */
471   _applyAttributeBindings: function(buffer) {
472     var attributeBindings = get(this, 'attributeBindings'),
473         attributeValue, elem, type;
474 
475     if (!attributeBindings) { return; }
476 
477     a_forEach(attributeBindings, function(binding) {
478       var split = binding.split(':'),
479           property = split[0],
480           attributeName = split[1] || property;
481 
482       // Create an observer to add/remove/change the attribute if the
483       // JavaScript property changes.
484       var observer = function() {
485         elem = this.$();
486         attributeValue = get(this, property);
487 
488         Ember.View.applyAttributeBindings(elem, attributeName, attributeValue);
489       };
490 
491       addObserver(this, property, observer);
492 
493       // Determine the current value and add it to the render buffer
494       // if necessary.
495       attributeValue = get(this, property);
496       Ember.View.applyAttributeBindings(buffer, attributeName, attributeValue);
497     }, this);
498   },

500   /**
501     @private
502 
503     Given a property name, returns a dasherized version of that
504     property name if the property evaluates to a non-falsy value.
505 
506     For example, if the view has property `isUrgent` that evaluates to true,
507     passing `isUrgent` to this method will return `"is-urgent"`.
508   */
509   _classStringForProperty: function(property) {
510     var split = property.split(':'),
511         className = split[1];
512 
513     property = split[0];
514 
515     // TODO: Remove this `false` when the `getPath` globals support is removed
516     var val = Ember.getPath(this, property, false);
517     if (val === undefined && Ember.isGlobalPath(property)) {
518       val = Ember.getPath(window, property);
519     }
520 
521     // If value is a Boolean and true, return the dasherized property
522     // name.
523     if (val === true) {
524       if (className) { return className; }
525 
526       // Normalize property path to be suitable for use
527       // as a class name. For exaple, content.foo.barBaz
528       // becomes bar-baz.
529       var parts = property.split('.');
530       return Ember.String.dasherize(parts[parts.length-1]);
531 
532     // If the value is not false, undefined, or null, return the current
533     // value of the property.
534     } else if (val !== false && val !== undefined && val !== null) {
535       return val;
536 
537     // Nothing to display. Return null so that the old class is removed
538     // but no new class is added.
539     } else {
540       return null;
541     }
542   },
